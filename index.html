packages = ["pandas", "xlsxwriter"] import pandas as pd import io from pyodide.http import open_url from js import Blob, document, URL async def executar(event): status = document.getElementById("status") status.innerHTML = "Processando..." try: # 1. BUSCA DOS DADOS (Convertendo a URL para garantir CSV limpo) url = "https://docs.google.com/spreadsheets/d/18yTaoaf6_dCcZSkHdUFVnreF_GjT5cZoShkpxbcfPG8/export?format=csv" # No PyScript, usamos open_url para ler a rede do navegador corretamente dados_csv = open_url(url) df = pd.read_csv(dados_csv) # Extraindo os valores (como você fez no seu código) data_obs = df['Data da observação'] telescopio = df['Telescópio'] observadores = df['Câmera'] # 2. CRIAÇÃO DO DF2 (Estrutura de posições) df2 = pd.DataFrame(index=range(5), columns=['A', 'B']) df2.at[1, 'B'] = data_obs[0] df2.at[2, 'B'] = telescopio[0] df2.at[3, 'B'] = observadores[0] # 3. EXPORTAÇÃO PARA MEMÓRIA (Substitui o arquivo físico no PC) output = io.BytesIO() file_name = f"relatorio_observacao_{str(data_obs[0]).replace('/','-')}.xlsx" writer = pd.ExcelWriter(output, engine='xlsxwriter') df2.to_excel(writer, index=False, header=False, sheet_name='Relatório') workbook = writer.book worksheet = writer.sheets['Relatório'] # Formatos e Mesclagem (Igual ao seu código) merge_format = workbook.add_format({'bold': 1, 'border': 1, 'align': 'center', 'valign': 'vcenter', 'fg_color': '#D7E4BC'}) worksheet.merge_range(0, 0, 0, 1, "Relatório da observação", merge_format) # Ajuste de largura (Igual ao seu código) worksheet.set_column('A:A', 20) lista_textos = [str(data_obs[0]), str(telescopio[0]), str(observadores[0])] maior_texto = max(len(t) for t in lista_textos) worksheet.set_column('B:B', maior_texto + 2) writer.close() output.seek(0) # 4. DOWNLOAD NO NAVEGADOR blob = Blob.new([output.getvalue()], {type: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"}) url_download = URL.createObjectURL(blob) link = document.createElement("a") link.href = url_download link.download = file_name link.click() status.innerHTML = "✅ Concluído!" except Exception as e: status.innerHTML = f"❌ Erro: {str(e)}"

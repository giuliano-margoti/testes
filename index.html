packages = ["pandas", "xlsxwriter", "openpyxl", "ssl"] import pandas as pd import io from pyodide.http import open_url from js import Uint8Array, File, URL, document, window # Avisar que o sistema está pronto document.getElementById("status").innerHTML = "✅ Pronto para gerar!" def gerar_relatorio(event): status = document.getElementById("status") status.innerHTML = "Processando... aguarde." try: # 1. Dados da Planilha url = "https://docs.google.com/spreadsheets/d/18yTaoaf6_dCcZSkHdUFVnreF_GjT5cZoShkpxbcfPG8/export?format=csv" df = pd.read_csv(open_url(url)) data_obs = str(df['Data da observação'].iloc[0]) telescopio = str(df['Telescópio'].iloc[0]) observadores = str(df['Câmera'].iloc[0]) df2 = pd.DataFrame(index=range(5), columns=['A', 'B']) df2.at[1, 'B'] = data_obs df2.at[2, 'B'] = telescopio df2.at[3, 'B'] = observadores # 2. Gerando o Excel em memória output = io.BytesIO() writer = pd.ExcelWriter(output, engine='xlsxwriter') df2.to_excel(writer, index=False, header=False, sheet_name='Relatório') workbook = writer.book worksheet = writer.sheets['Relatório'] merge_format = workbook.add_format({'bold': 1, 'border': 1, 'align': 'center', 'valign': 'vcenter', 'fg_color': '#D7E4BC'}) worksheet.merge_range(0, 0, 0, 1, "Relatório da observação", merge_format) worksheet.set_column('A:A', 20) worksheet.set_column('B:B', 35) writer.close() # 3. Truque para baixar o arquivo no navegador via JS processed_data = output.getvalue() file_name = f"relatorio_{data_obs.replace('/','-')}.xlsx" js_array = Uint8Array.new(processed_data) file = File.new([js_array], file_name, {type: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"}) url = URL.createObjectURL(file) hidden_link = document.createElement("a") hidden_link.href = url hidden_link.download = file_name document.body.appendChild(hidden_link) hidden_link.click() document.body.removeChild(hidden_link) status.innerHTML = "✅ Download concluído!" except Exception as e: status.innerHTML = f"❌ Erro: {str(e)}"

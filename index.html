Relatório de Observação

Clique no botão abaixo para ler os dados da planilha e gerar o arquivo Excel.

packages = ["pandas", "xlsxwriter", "openpyxl"] import pandas as pd import io from js import Blob, document, URL # Integração com o JavaScript do navegador from pyodide.http import open_url # Importe isso no topo do seu async def gerar_relatorio(event): status = document.getElementById("status") status.innerHTML = "⌛ Lendo planilha do Google e gerando arquivo..." try: # 1. Definição da URL da planilha (exportando como CSV) url = "https://docs.google.com/spreadsheets/d/18yTaoaf6_dCcZSkHdUFVnreF_GjT5cZoShkpxbcfPG8/edit?usp=sharing" # Usamos o open_url para contornar restrições de rede do navegador csv_data = open_url(url) df_origem = pd.read_csv(csv_data) # 3. Extraindo as informações (pegando o primeiro registro [0]) data_obs = str(df_origem['Data da observação'][0]) telescopio = str(df_origem['Telescópio'][0]) observadores = str(df_origem['Câmera'][0]) # 4. Criando o df2 com a estrutura de posições específicas (B2, B3, B4) df2 = pd.DataFrame(index=range(5), columns=['A', 'B']) df2.at[1, 'B'] = data_obs # Posição 2B df2.at[2, 'B'] = telescopio # Posição 3B df2.at[3, 'B'] = observadores # Posição 4B # 5. Configurando a escrita do Excel em memória (buffer) output = io.BytesIO() writer = pd.ExcelWriter(output, engine='xlsxwriter') df2.to_excel(writer, index=False, header=False, sheet_name='Relatório') # 6. Formatação Visual workbook = writer.book worksheet = writer.sheets['Relatório'] # Criando estilo para a célula mesclada (verde claro) merge_format = workbook.add_format({ 'bold': 1, 'border': 1, 'align': 'center', 'valign': 'vcenter', 'fg_color': '#D7E4BC' }) # Mesclando A1 e B1 (índice 0,0 até 0,1) worksheet.merge_range(0, 0, 0, 1, "Relatório da observação", merge_format) # Ajustando larguras: Coluna A fixa em 20, Coluna B dinâmica worksheet.set_column('A:A', 20) largura_b = max(len(data_obs), len(telescopio), len(observadores)) + 5 worksheet.set_column('B:B', largura_b) # Fecha o arquivo Excel na memória writer.close() output.seek(0) # 7. Lógica de Download no Navegador nome_arquivo = f"relatorio_{data_obs.replace('/', '-')}.xlsx" file_data = output.getvalue() # Transforma os bytes do Python em um objeto de download do navegador blob = Blob.new([file_data], {type: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"}) url_download = URL.createObjectURL(blob) # Cria um link invisível e "clica" nele para baixar link = document.createElement("a") link.href = url_download link.download = nome_arquivo link.click() status.innerHTML = f"✅ Sucesso! Arquivo '{nome_arquivo}' baixado." except Exception as e: status.innerHTML = f"❌ Erro: Verifique se a planilha é pública.
{str(e)}"

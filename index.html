Smith" df2 = pd.DataFrame(index=range(5), columns=['A', 'B']) df2.at[1, 'B'] = data_obs df2.at[2, 'B'] = telescopio df2.at[3, 'B'] = observadores # 2. Criar o Excel em memória (BytesIO) output = io.BytesIO() writer = pd.ExcelWriter(output, engine='xlsxwriter') df2.to_excel(writer, index=False, header=False, sheet_name='Relatório') # Formatação (mesmo código que já usamos) workbook = writer.book worksheet = writer.sheets['Relatório'] merge_format = workbook.add_format({'bold': 1, 'border': 1, 'align': 'center', 'valign': 'center', 'fg_color': '#D7E4BC'}) worksheet.merge_range(0, 0, 0, 1, "Relatório da observação", merge_format) worksheet.set_column('B:B', 40) writer.close() output.seek(0) # 3. Criar o link de download no navegador file_data = output.getvalue() blob = Blob.new([file_data], {type: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"}) url = URL.createObjectURL(blob) link = document.createElement("a") link.href = url link.download = "relatorio_observacao.xlsx" link.click() status.innerHTML = "Download concluído!"
